macro(add_pixel_test name)
  add_wn_executable(${name}
    SOURCES
      ${PIXEL_TEST_SOURCES}
    INCLUDES
      ${WNFramework_SOURCE_DIR}/externals/stb
    LIBS
      executable_test
      WNGraphics
      WNFileSystem
      stb
  )

  wn_create_call_test(
    ${name}
      WORKING_DIRECTORY
        ${WNFramework_BINARY_DIR}
      APPLICATION
        ${name}
      COMMAND_PRE_DIR
        -image_path
      COMMAND_DIR ${CMAKE_CURRENT_SOURCE_DIR}/golden_images
  )
endmacro()
